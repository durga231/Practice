

CREATE TABLE  "CITY_CLASSIFIED" 
   (	"ID" NUMBER, 
	"BUYSELL" VARCHAR2(4000), 
	"NAME" VARCHAR2(4000), 
	"DESCRIPTION" VARCHAR2(4000), 
	"USERID" NUMBER, 
	 CONSTRAINT "CITY_CLASSIFIED_PK" PRIMARY KEY ("ID") ENABLE
   )
/

CREATE OR REPLACE TRIGGER  "BI_CITY_CLASSIFIED" 
  before insert on "CITY_CLASSIFIED"               
  for each row  
begin   
    select "CITY_CLASSIFIED_SEQ".nextval into :NEW.ID from dual; 
end; 

/
ALTER TRIGGER  "BI_CITY_CLASSIFIED" ENABLE
/
